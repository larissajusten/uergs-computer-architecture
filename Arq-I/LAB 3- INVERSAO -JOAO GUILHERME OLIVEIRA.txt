MOV BL,C0
	CLO
INICIO:
	INC CL
	IN 00
	CMP AL,0D
	JZ INVERSAO
	CMP AL,60
	JNS MENOR
COMPARA:
	CMP BL,D0
	JZ LN
IMPRIME:
	MOV [BL],AL
	INC BL
	JMP INICIO
MENOR: 
	SUB AL,20
	JMP COMPARA
LN:
	MOV BL,C0
	JMP IMPRIME


INVERSAO:
	MOV CL,CF

PRINT:
	MOV AL,[BL]
	MOV [CL],AL
	DEC BL
	CMP BL,BF
	JZ FINAL
	INC CL
	JMP PRINT
FINAL:

END
